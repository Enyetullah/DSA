"""
Contains Duplicate
Given an integer array nums, return true if any value appears more than once in the array, otherwise return false.

Example 1:

Input: nums = [1, 2, 3, 3]

Output: true

Example 2:

Input: nums = [1, 2, 3, 4]

Output: false

"""

class Solution:
    def hasDuplicate(self, nums: List[int]) -> bool:
        # Create a set so that numbers are stored
        seen = set()
        # Check to see if the current element is already added to the set
        for num in nums:
            if num in seen:
                return True
            seen.add(num)
        return False
